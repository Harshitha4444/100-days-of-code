import java.util.*;
public class Main
{
  public static void main(String[] args) 
   {
       Scanner sc=new Scanner(System.in);
       int n=sc.nextInt();
       int temp=n;
       int d=0;
       while(temp!=0)
       {
           temp=temp/10;
           d++;
       }
       int div=(int)Math.pow(10,d-1);
       while(div!=0)
       {
          int h=n/div;
          System.out.println(h);
          n=n%div;
          div=div/10;
       }
    }
}
